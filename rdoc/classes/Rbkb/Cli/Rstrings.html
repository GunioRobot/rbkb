28:     arg.on("-s", "--start=OFFSET", "Start at offset") do |s|
29:       unless m=/^(?:(\d+)|0x([A-Fa-f0-9]+))$/.match(s)
30:         bail "invalid offset '#{s}'"

33:     end
34: 
35:     arg.on("-e", "--end=OFFSET", "End at offset") do |e|

37:         bail "invalid offset '#{e}'"
